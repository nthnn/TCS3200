404 template <typename T>
405 T nearest_color(T *color_labels, RGBColor *color_values, int size) {
406 T nearest;
407
408 RGBColor readings = this->read_rgb_color();
409 uint16_t min_dist = 0xffff;
410
411 for(int i = 0; i < size; i++) {
412 uint16_t dist = abs(readings.red - color_values[i].red) +
413 abs(readings.green - color_values[i].green) +
414 abs(readings.blue - color_values[i].blue);
415
416 if(dist < min_dist) {
417 min_dist = dist;
418 nearest = color_labels[i];
419 }
420 }
421
422 return nearest;
423 }
