515 template <typename T>
516 T nearest_color(T *color_labels, RGBColor *color_values, int size) {
517 T nearest;
518
519 RGBColor readings = this->read_rgb_color();
520 uint16_t min_dist = 0xffff;
521
522 for(int i = 0; i < size; i++) {
523 uint16_t dist = abs(readings.red - color_values[i].red) +
524 abs(readings.green - color_values[i].green) +
525 abs(readings.blue - color_values[i].blue);
526
527 if(dist < min_dist) {
528 min_dist = dist;
529 nearest = color_labels[i];
530 }
531 }
532
533 return nearest;
534 }
